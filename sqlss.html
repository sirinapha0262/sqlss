1.1 ความหมายและประวัติ SQL พื้นฐาน ▶
SQL (Structured Query Language) คือภาษามาตรฐานสำหรับจัดการ ฐานข้อมูลเชิงสัมพันธ์ (Relational Database) พัฒนาโดย IBM ในปี 1970 เป็นมาตรฐาน ISO/ANSI

ใช้กับ: MySQL · MariaDB · PostgreSQL · SQLite · Oracle · MS SQL Server

ในสไลด์: ใช้ MariaDB 10.4 ผ่าน XAMPP

DCLGRANT, REVOKEจัดการสิทธิ์ผู้ใช้

CREATE · ALTER · DROP

คำสั่งสร้างและจัดการโครงสร้างฐานข้อมูล

4.1 CREATE TABLE — สร้างตาราง จากสไลด์ ▶
ตารางโปรแกรมวิชา (department)
 SQL — จากสไลด์
CREATE TABLE department (
  iddpm    INT(1) PRIMARY KEY,
  dpm_name VARCHAR(15)
);
-- Query OK, 0 rows affected (0.821 sec)
ตารางนักศึกษา (student) — มี Foreign Key
CREATE TABLE student (
  id      INT(3) PRIMARY KEY,
  name    VARCHAR(15),
  address VARCHAR(20),
  iddpm   INT(1),
  CONSTRAINT fk_iddpm FOREIGN KEY(iddpm)
  REFERENCES department(iddpm)
  ON UPDATE CASCADE
);
-- Query OK, 0 rows affected (0.328 sec)
✅
ON UPDATE CASCADE = เมื่อค่า PRIMARY KEY ใน department เปลี่ยน FOREIGN KEY ใน student จะอัปเดตตามอัตโนมัติ

SELECT — ดึงข้อมูล

คำสั่งที่ใช้บ่อยที่สุดใน SQL

5.1 โครงสร้าง SELECT ครบสูตร syntax ▶
 โครงสร้างเต็ม
SELECT [DISTINCT] col1, col2, ...
FROM table_name
[WHERE condition]
[GROUP BY column]
[HAVING condition]
[ORDER BY column [ASC|DESC]]
[LIMIT n];
💡
ลำดับการเขียน: SELECT → FROM → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT
5.2 ตัวอย่าง SELECT จากสไลด์ครบทุกแบบ จากสไลด์ ▶
1. ดูทุกคอลัมน์
SELECT * FROM department;
iddpmdpm_name

2. เลือกบางคอลัมน์

3. WHERE กรองตรงๆ
SELECT id,name,address FROM student
WHERE address='Phangkhon';
4. WHERE + LIKE
SELECT * FROM student
WHERE name LIKE 'p%';
-- % = ตัวอักษรใดก็ได้
5. ORDER BY
SELECT id,name FROM student
ORDER BY name;
-- DESC = มากไปน้อย
6. COUNT นับจำนวน
SELECT COUNT(id) FROM student;
5.3 SELECT คุณสมบัติพิเศษ เพิ่มเติม ▶

FROM student;

IS NULL

Aggregate Functions

ฟังก์ชันสรุปข้อมูล: COUNT · SUM · AVG · MAX · MIN

11.1 ฟังก์ชัน Aggregate ทั้งหมด ▶
ฟังก์ชันทำอะไรตัวอย่าง
COUNT(col)นับแถวที่ไม่เป็น NULLSELECT COUNT(id) FROM student;
COUNT(*)นับทุกแถวรวม NULLSELECT COUNT(*) FROM student;
SUM(col)รวมค่าทั้งหมดSELECT SUM(score) FROM grade;
AVG(col)ค่าเฉลี่ยSELECT AVG(score) FROM grade;
MAX(col)ค่าสูงสุดSELECT MAX(score) FROM grade;
MIN(col)ค่าต่ำสุดSELECT MIN(score) FROM grade;
ตัวอย่างจากสไลด์
SELECT COUNT(id) FROM student;
-- ผล: 4
รวมหลาย Aggregate
SELECT
  MAX(score) AS คะแนนสูงสุด,
  MIN(score) AS คะแนนต่ำสุด,
  AVG(score) AS คะแนนเฉลี่ย,
  SUM(score) AS คะแนนรวม
FROM grade;

GROUP BY & HAVING

จัดกลุ่มและกรองผลลัพธ์ Aggregate

12.1 GROUP BY + HAVING ▶
-- นับนักศึกษาในแต่ละสาขา
SELECT iddpm, COUNT(id) AS จำนวน
FROM student
GROUP BY iddpm;

-- รวม JOIN ดูชื่อสาขา
SELECT d.dpm_name, COUNT(s.id) AS จำนวน
FROM student s
INNER JOIN department d ON s.iddpm=d.iddpm
GROUP BY d.dpm_name;

-- HAVING: กรองกลุ่มที่มีนักศึกษา > 1 คน
SELECT iddpm, COUNT(id) AS จำนวน
FROM student
GROUP BY iddpm
HAVING COUNT(id) > 1;
💡
WHERE vs HAVING: WHERE กรองแถวก่อน GROUP BY / HAVING กรองกลุ่มหลัง GROUP BY — ไม่สามารถใช้ aggregate function ใน WHERE ได้

Constraints & Keys

ข้อจำกัดและกฎที่กำหนดให้คอลัมน์

13.1 Constraints ทั้งหมด ▶
Constraintความหมายตัวอย่าง
PRIMARY KEYระบุตัวตน ไม่ซ้ำ ไม่ NULLid INT PRIMARY KEY
FOREIGN KEYอ้างอิง PK ตารางอื่นFOREIGN KEY(iddpm) REFERENCES department(iddpm)
UNIQUEไม่ซ้ำ (NULL ได้)email VARCHAR(50) UNIQUE
NOT NULLต้องมีค่าname VARCHAR(15) NOT NULL
DEFAULTค่าเริ่มต้นstatus VARCHAR(10) DEFAULT 'active'
CHECKตรวจสอบเงื่อนไขage INT CHECK(age >= 0)
AUTO_INCREMENTเพิ่มอัตโนมัติid INT AUTO_INCREMENT PRIMARY KEY
ตัวอย่างใช้หลาย Constraints
CREATE TABLE users (
  id       INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(30) NOT NULL UNIQUE,
  email    VARCHAR(50) UNIQUE,
  age      INT CHECK(age >= 0),
  status   VARCHAR(10) DEFAULT 'active'
);

FROM student

ON student.iddpm=department.iddpm
WHERE dpm_name='Electronic';
